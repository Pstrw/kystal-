	function damagealll(Radius,Position)		
		local Returning = {}		
		for _,v in pairs(workspaceGetChildren()) do		
			if v~=Character and vFindFirstChildOfClass('Humanoid') and vFindFirstChild('Torso') or vFindFirstChild('UpperTorso') then
	if vFindFirstChild(Torso) then		
				local Mag = (v.Torso.Position - Position).magnitude		
				if Mag  Radius then		
					table.insert(Returning,v)		
				end
	elseif vFindFirstChild(UpperTorso) then	
				local Mag = (v.UpperTorso.Position - Position).magnitude		
				if Mag  Radius then		
					table.insert(Returning,v)		
				end
	end	
			end		
		end		
		return Returning		
	end
